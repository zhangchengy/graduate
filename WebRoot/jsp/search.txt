<script type="text/javascript">

$(document).ready(function () {
					$("#search_submit").click(
						function () {
				$.ajax({
					type : "POST",
					url : "goodsfind",
					data : {
						search: $.trim($("#search").val()),

					},
					
					success : function(data) {
						location.href="jsp/result.jsp";

					},
					error : function(jqXHR) {
						alert("发生错误：" + jqXHR.status);
					}
				});
			});
});
